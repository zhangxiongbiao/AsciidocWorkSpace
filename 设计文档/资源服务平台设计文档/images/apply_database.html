Table S_SERVICE_RESOURCE
服务资源信息
  ID integer 主键ID
* FWBS varchar2 服务标识
  FWMC varchar2 服务名称
  FW_MS varchar2 服务_描述 关于服务的功能、特性、使用方式等信息的说明性文字；
  FWLXBM varchar2 服务类型编码 应符合GA/T XX2-201X中7.7的规定；
  FW_XZQHDM varchar2 服务_行政区划代码。 服务所提供资源覆盖的辖区范围
  FW_MMDJDM varchar2 服务_秘密等级代码； 服务所提供资源的涉密级别；
  TGZ_ZDSYFW varchar2 提供者_制定使用范围 描述提供者制定的允许访问该服务的请求者范围，包括请求者的： 1) 行政范围。职务级别代码应符合4.3的规定； 2) 岗位范围。岗位权级代码应符合4.3的规定； 3) 辖区范围。行政区划代码应符合4.3的规定； 4) 涉密范围。秘密等级代码应符合4.3的规定； 5) 警种范围。警种（部门）代码应符合4.3的规定。 以及不同权级对应获得的不同结果集。应符合第5章的规定；
  ZX_ZDSYFW varchar2 总线_制定使用范围 描述总线制定的允许访问该服务的请求者范围，包括请求者的： 1) 行政范围。职务级别代码应符合4.3的规定； 2) 岗位范围。岗位权级代码应符合4.3的规定； 3) 辖区范围。行政区划代码应符合4.3的规定； 4) 涉密范围。秘密等级代码应符合4.3的规定； 5) 警种（部门）范围。警种（部门）代码应符合4.3的规定。 以及不同权级对应获得的不同结果集。应符合第5章的规定；
  FWTZ_ RQSJ date 服务停止_日期时间
  FWZC_RQSJ date 服务注册_日期时间
  FWTGZ_GAJGJGDM varchar2 服务提供者_标识 用于唯一标识一个服务提供者，即服务提供者的公安机关机构代码；
  FWZTDM varchar2 服务状态代码。 服务的可用性状态。应符合第6章的规定；
  FWDZ_ID varchar2 服务地址ID
  PUBLISH_STATUS varchar2 服务发布状态00：下线，01：上线
  AUDIT_STATUS varchar2 服务审批状态（00：未审核，01：审核中，02：审核通过，03：审核不通过）
  METHOD_NAME varchar2 服务的方法名称
  REQUEST_EXAMPLE clob 服务请求报文示例
  RESPONSE_EXAMPLE clob 服务响应报文示例
Indexes
