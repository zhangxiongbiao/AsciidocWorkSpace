  MMDJDM varchar2 秘密等级代码
  JZDM varchar2 警种代码（部门）
  FWQQZRKDZLB varchar2 服务请求者入口地址列表
  FWQQZ_IPDZ varchar2 服务请求者入口地址
  FWQQZ_ZXDKHM integer 服务请求者_端口号
  FWQQZ_ZXBS varchar2 服务请求者总线标识 索引字段
  FWQQZ_MS varchar2 服务请求者_描述
  FWQQZ_LXFS varchar2 服务请求者_联系方式
  LXR_SM varchar2 联系人_说明
  LXR_XM varchar2 联系人_姓名
  LXDH varchar2 联系电话
  DZXX varchar2 电子信箱
  TXDZ varchar2 通信地址
  LXR_GMSFHM varchar2 联系人_身份证号
  STATUS integer 状态（0不可用，1可用，2删除）
Indexes
PkPk_S_SERVICE_CONSUMER ON ID
 idx_S_SERVICE_CONSUMER ON FWQQZ_ZXBS
Foreign Keys
 fk_S_SERVICE_CONSUMER ( FWQQZ_ZXBS ) ref S_SERVICE_BUS (ZXBS)
